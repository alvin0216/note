<style>
  .container {
    display: grid;
    grid-template-columns: 100px 100px;
    grid-row: auto auto;
    grid-column-gap: 20px;
    grid-row-gap: 20px;
  }

  .box {
    height: 100px;
    background: yellow;
  }

  .box3 {
    transform: scaleX(-1);
  }

  .box2 {
    transform: scaleY(-1);
  }

  .box1 {
    transform: scale(-1);
  }
</style>

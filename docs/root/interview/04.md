使用 [二分查找](../../algorithm/temp/binarySearch.md) 完成

```js
function findNext(target, arr) {
  let min = 0,
    max = arr.length - 1

  while (min <= max) {
    const mid = (max + min) >>> 1 // 位运算，无符号右移一位，同 Math.floor((min + max) / 2)
    if (target > arr[mid]) {
      min = mid + 1
    } else if (target < arr[mid]) {
      max = mid - 1
    } else {
      return arr[mid]
    }
  }

  return arr[min] - target <= target - arr[max] ? arr[min] : arr[max]
}
```

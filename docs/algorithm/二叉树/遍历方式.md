- 中序遍历（左根右）：`1 2 3 4`
- 前序遍历（根左右）：`3 1 2 4`
- 后序遍历（左右根）：`2 1 4 3`

## DFS

中序遍历

```js
//    3
//   / \
//  1   4
//   \
//    2
// 输出: 1 2 3 4
function dfs(node) {
  if (!node) return;
  node.left && dfs(node.left);
  console.log(node.val);
  node.right && dfs(node.right);
}
```

栈版本

```js
function dfs(root) {
  const stack = [];
  let nums = [];
  let current = root;

  while (current || stack.length > 0) {
    while (current) {
      stack.push(current);
      current = current.left;
    }
    current = stack.pop();
    nums.push(current.val);
    current = current.right;
  }

  console.log(nums); // [ 1, 2, 3, 4 ]
}
```

## BFS

```js
//    3
//   / \
//  1   4
//   \
//    2
// 输出: 3 1 4 2
function bfs(root) {
  let queue = [root];

  while (queue.length) {
    const node = queue.shift();
    console.log(node.val); //
    node.left && queue.push(node.left);
    node.right && queue.push(node.right);
  }
}
```

/**
 * @param {string} s
 * @return {number}
 */
var lengthOfLongestSubstring = function (s) {
  let start = 0,
    end = 0,
    max = 0,
    map = new Map();

  while (end < s.length) {
    if (map.has(s[end])) {
      start = Math.max(start, map.get(s[end]) + 1);
    }
    map.set(s[end], end);
    end++;
    max = Math.max(max, end - start);
  }

  return max;
};

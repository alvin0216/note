    function binarySearch(arr, maxIndex, value) {
  let min = 0;
  let max = maxIndex;

  while (min <= max) {
    const mid = Math.floor((min + max) / 2);

    if (arr[mid] <= value) {
      min = mid + 1;
    } else {
      max = mid - 1;
    }
  }

  return min;
}

    console.log('--------'); // getItems return items

    towerOfHanoi(n - 1, help, from, to); // B -> C
  }
};

function hanoiStack(plates) {
  const from = new Stack(),
    help = new Stack(),
    to = new Stack();

  for (let i = plates; i > 0; i--) {
    from.push(i);
  }

  towerOfHanoi(plates, from, help, to);
}

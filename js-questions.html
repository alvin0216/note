    # 函数声明 & 函数提升

    # 输出是什么？

    fn();

function fn() {
  console.log(1);
}

fn();
function fn() {
  console.log(2);
}
    答案

    答案: 2, 2 函数声明覆盖了。

    # 输出是什么？

    fn();

var fn = function () {
  console.log(1);
};

fn();
function fn() {
  console.log(2);
}
    答案

    答案: 2 1, 第一个 fn 函数提升，第二个 fn 被 var 函数重声明了

    # 输出是什么？

    var length = 10;
function func() {
  console.log(this.length);
}
var obj = {
  length: 5,
  run: function (func) {
    func();
    arguments[0]();
  },
};

obj.run(func, 123);
    答案

    • func() 执行环境是全局。length = 10
    • arguments[0]()：arguments 也是对象，也有 length 属性. length = 2

    # 输出是什么？

    var name = 'global';

var obj = {
  name: 'obj',
  a() {
    return function () {
      console.log(this.name);
    };
  },
};

obj.a()();
    答案

    global

    obj.a()() 执行环境是 obj.a() 也就是全局。

    # 输出是什么？

    var name = 'windowsName';

var a = {
  name: 'Cherry',
  func1: function () {
    console.log(this.name);
  },
  func2: function () {
    setTimeout(function () {
      console.log(this.name);
      this.func1();
    }, 100);
  },
};

a.func2();
    答案 

    var name = 'windowsName';

var a = {
  name: 'Cherry',
  func1: function () {
    console.log(this.name);
  },
  func2: function () {
    setTimeout(function () {
      console.log(this.name); // windowsName
      this.func1(); // this.func1 is not a function
    }, 100);
  },
};

a.func2();

    this 的运行时是在一个定时器中，因为它是匿名函数被单独调用，其上下文是全局，所以没有 func1 这个方法。所以必然报错。

    # 输出是什么？

    var a = {
  b: function () {
    console.log(this);
  },
  c: () => {
    console.log(this);
  },
  d: {
    e: () => {
      console.log(this);
    },
    g: {
      i: () => {
        console.log(this);
      },
    },
  },
  f: function () {
    var func = () => {
      console.log(this);
    };
    func();
  },
};

a.b();
a.c();
a.d.e();
a.d.g.i();
a.f();
    答案 

    a.b(); // a
a.c(); // window
a.d.e(); // window
a.d.g.i(); // window
a.f(); // a
    1. 箭头函数里面的 this 就是定义时上层作用域中的 this！
    2. 注意 this 指的是函数运行时所在的环境，a 对象并没有被函数包裹，所以输出 window 是默认只想上一级环境，也就是 window
